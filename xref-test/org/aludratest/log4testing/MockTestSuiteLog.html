11  
12  public class MockTestSuiteLog extends MockNamedTestLogElement implements TestSuiteLog {
13  
14  	private TestSuiteLog parent;
15  
16  	private List<TestSuiteLog> childSuites = new ArrayList<TestSuiteLog>();
17  
18  	private List<TestCaseLog> testCases = new ArrayList<TestCaseLog>();
19  
20  	public MockTestSuiteLog(int id, String name) {
21  		super(id, name);
22  	}
23  
24  	public MockTestSuiteLog(int id, String name, TestSuiteLog parent) {
25  		super(id, name);
26  		this.parent = parent;
27  	}
28  
29  	@Override
30  	public TestSuiteLog getParent() {
31  		return parent;
32  	}
33  
34  	@Override
35  	public List<? extends TestSuiteLog> getChildSuites() {
36  		return childSuites;
37  	}
38  
39  	@Override
40  	public List<? extends TestCaseLog> getTestCases() {
41  		return testCases;
42  	}
43  
44  	@Override
45  	public TestSuiteStatistics gatherStatistics() {
46  		return DefaultTestSuiteStatistics.create(this);
47  	}
48  
49  	public void addTestSuite(TestSuiteLog suite) {
50  		childSuites.add(suite);
51  	}
52  
53  	public void addTestCase(TestCaseLog testCase) {
54  		testCases.add(testCase);
55  	}
56  
57  	@Override
58  	public TestStatus getStatus() {
59  		TestStatus status = TestStatus.PASSED;
60  		for (TestCaseLog testCase : testCases) {
61  			if (testCase.getStatus().ordinal() < status.ordinal()) {
62  				status = testCase.getStatus();
63  			}
64  		}
65  
66  		return status;
67  	}
68  }
