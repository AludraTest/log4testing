3   import java.util.Properties;
4   
5   import org.aludratest.log4testing.TestStepFilter;
6   import org.aludratest.log4testing.TestStepLog;
7   import org.aludratest.log4testing.config.InvalidConfigurationException;

9   /**
10   * A very simple test step filter implementation which excludes ALL test steps from being logged. This is useful e.g. for XML
11   * generation where only statistics down to test case level are required.
12   * 
13   * @author falbrech
14   * 
15   */
16  public class NoTestStepsFilter implements TestStepFilter {
17  
18  	@Override
19  	public void init(Properties properties) throws InvalidConfigurationException {
20  		// nothing do do for a no-sayer
21  	}
22  
23  	@Override
24  	public boolean filter(TestStepLog testStep) {
25  		// NO steps, please
26  		return false;
27  	}
28  
29  }
